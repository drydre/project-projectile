90 public static void CheckClick(ref uint displaying, Loop loop)

92 Area area = MouseEventsMenu.CheckArea(loop);

152 public static void DrawMenu(Loop loop)
153 { // wyświelanie tytułu gry
154 loop.Window.Draw(titleText);
155 foreach(Miniature min in miniatures)
156 { // zmiana koloru ramki w zależności od ukończenia poziomu
157 min.active = false;
158 if (min.IsRect())
159 {
160 if (min.completed == true && min.rectShp.OutlineColor != cpltBrdClr)
161 min.ChangeColor(cpltBrdClr);
162 if (min.completed == false && min.rectShp.OutlineColor == cpltBrdClr)
163 min.ChangeColor(defBrdClr);
164 }
165 }
166 CheckArea(loop);
167 foreach (Miniature min in miniatures)
168 { // zmiana koloru w zależności od obecności kursora w obszarze
169 if (min.active == true)
170 {
171 if (min.IsRect())
172 {
173 min.ChangeColor(Color.White);
174 min.rectShp.FillColor = activeFillClr;
175 }
176 else
177 {
178 min.ellipse.ChangeEllipseColor(Color.White);
179 min.ellipse.ChangeFillColor(activeRstClr);
180 }
181 }
182 else if(min.IsRect())
183 { // powrót do standardowych kolorów z podświetlenia dla miniatur poziomów
184 if (min.active == false && (min.rectShp.OutlineColor == Color.White || min.rectShp.OutlineColor == cpltBrdClr))
185 {
186 if (min.completed == false)
187 min.ChangeColor(defBrdClr);
188 else
189 min.ChangeColor(cpltBrdClr);
190 min.rectShp.FillColor = Color.Transparent;
191 }
192 }
193 else if(!min.IsRect())
194 { // powrót do standardowych kolorów z podświetlenia dla przycisku resetu postępu
195 if (min.active == false && min.ellipse.GetOutlineColor() == Color.White)
196 {
197 min.ellipse.ChangeEllipseColor(defBrdClr);
198 min.ellipse.ChangeFillColor(rstClr);
199 }
200 }
201 // wyświetlenie każdej miniatury z listy
202 min.Draw(loop);
203 }
204 }
