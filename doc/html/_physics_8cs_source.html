25 public void InitShot(float alpha, float E, float m, Vector2f startCoords, float g)
26 { // inicjalizacja strzału - przekazanie obiektowi wszystkich potrzebnych
27 // danych do wykonania obliczeń
28 this.alpha = alpha;
29 this.E = E;
30 this.m = m;
31 this.startCoords = startCoords;
32 double V0 = GetV0(this.E, this.m);
33 // rozkład prędkości wylotowej na składowe X i Y
34 this.V0x = Math.Cos(alpha) * V0;
35 this.V0y = Math.Sin(alpha) * V0;
36 this.g = g * C.DISTANCE_SCALE;
37 // uruchomienie odliczania czasu od wylotu pocisku z procy
38 this.clock = new Clock();
39 }

41 public Vector2f UpdateCoords()
42 { // metoda zwracająca koordynaty położenia pocisku
43 // w zależności od upłyniętego czasu
44 float t = clock.ElapsedTime.AsSeconds();
45 float x = startCoords.X + (float)V0x * t;
46 //znak odwrócony żeby koordynaty były względem góry
47 float y = startCoords.Y - (float)V0y * t + g * (float)Math.Pow(t, 2)/2;
48 Vector2f coords = new Vector2f(x, y);
49 return coords;
50 }
51
52 public float UpdateVelocity()
53 { // metoda zwracająca prędkość pocisku
54 // wyświetlaną w trakcie jego lotu
55 float t = clock.ElapsedTime.AsSeconds();
56 double Vy = V0y - g * t;
57 float v = (float)Math.Sqrt(Math.Pow(V0x, 2) + Math.Pow(Vy, 2));
58 return v;
59 }

61 public double GetV0(float E, float m)
62 { // metoda zwracająca prędkość początkową
63 // pocisku, która jest następnie rozkładana
64 // na składowe poziomą i pionową
65 double V0 = Math.Sqrt(2 * E / m)*C.DISTANCE_SCALE;
66 return V0;
67 }

69 public float GetEnergy(float stringK, float stretch)

75 // POTRZEBNEJ DO UNIESIENIA POCISKU DO PUNKTU WYLOTU
76 float energy = (float)(stringK * Math.Pow(stretch/C.DISTANCE_SCALE, 2) / 2);
