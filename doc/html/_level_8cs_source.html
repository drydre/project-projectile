215 public void CheckClickUp(ref uint displaying, Loop loop)
216 {
217 Area area = MouseEventsGame.CheckAreaUp(this, stage, loop);
218 if (area != Area.none && area != Area.projectile)
219 {
220 click.Play();
221 }
222
223 if (area == Area.projectile && activity == Action.aiming)
224 { // głośność strzału jest zależna od naciągnięcia cięciwy
225 // oraz od planety, na której rozgrywany jest poziom
226 // (z uwagi na różnice sił naciągu pomiędzy poziomami)
227 shot.Volume = 100*(float)Math.Sqrt(g/C.G_KEPLER) * stage.energy / stage.maxEnergy;
228 shot.Play();
229 }
230
231 switch (area)
232 { // w zależności od obszaru, w którym nastąpiło zwolnienie
233 // przycisku myszy, podejmowane są różne czynności
234 case Area.projectile:
235 if (activity == Action.aiming)
236 activity = Action.acc;
237 break;
238 case Area.theory:
239 if (activity == Action.theory || activity == Action.none)
240 {
241 this.displayTheory = !this.displayTheory;
242 if (displayTheory)
243 activity = Action.theory;
244 else
245 activity = Action.none;
246 }
247 break;
248 case Area.leftButton:
249 if (activity == Action.result)
250 {
251 displaying = 0;
252 activity = Action.none;
253 }
254 break;
255 case Area.rightButton:
256 if (activity == Action.result)
257 {
258 activity = Action.none;
259 this.LoadStage();
260 }
261 break;
262 case Area.none:
263 break;
264 default:
265 break;
266 }
267 }

288 public void CheckMove(Loop loop)
289 {
290 Area area = MouseEventsGame.CheckAreaUp(this, stage, loop);

306 public void UpdateLevel(Loop loop)
307 {
308 if (MouseEventsGame.CheckArea(loop) == Area.theory)
309 this.ellipse.ChangeFillColor(DisplayLevel.theoryFillClrActive);
310 else
311 this.ellipse.ChangeFillColor(DisplayLevel.theoryFillClr);
312 // warunek spełniony, gdy istnieje rezultat strzału
313 if (stage.updatedResult != 0 && stage.stageNumber == currentStage)
314 {
315 if (stage.updatedResult == 1)
316 { // zwiększanie numeru etapu w razie sukcesu
317 // jeśli ostatni poziom, to zapisz postęp w pliku
318 if (stage.stageNumber == 3)
319 Progress.SaveProgress((short)levelNumber);
320 currentStage = stage.stageNumber + 1;
321 }
322 activity = Level.Action.result;
323 }
324 // jeśli brak jeszcze wyniku, to odśwież obliczenia charakterystyczne dla etapu
325 else if (!(activity == Action.result && stage.updatedResult == 0))
326 stage.UpdateStage(ref activity, loop);
