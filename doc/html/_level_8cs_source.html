158 public void CheckClickUp(ref uint displaying, Loop loop)
159 { // metoda uruchamiana przy zdarzeniu zwolnienia przycisku myszy
160 Area area = MouseEventsGame.CheckAreaUp(this, stage, loop);
161 if (area != Area.none && area != Area.projectile)
162 {
163 click.Play();
164 }
165
166 if (area == Area.projectile && activity == Action.aiming)
167 { // głośność strzału jest zależna od naciągnięcia cięciwy
168 // oraz od planety, na której rozgrywany jest poziom
169 // (z uwagi na różnice sił naciągu pomiędzy poziomami)
170 shot.Volume = 100*(float)Math.Sqrt(g/C.G_KEPLER) * stage.energy / stage.maxEnergy;
171 shot.Play();
172 }
173
174 switch (area)
175 { // w zależności od obszaru, w którym nastąpiło zwolnienie
176 // przycisku myszy, podejmowane są różne czynności
177 case Area.projectile:
178 if (activity == Action.aiming)
179 activity = Action.acc;
180 break;
181 case Area.theory:
182 if (activity == Action.theory || activity == Action.none)
183 {
184 this.displayTheory = !this.displayTheory;
185 if (displayTheory)
186 activity = Action.theory;
187 else
188 activity = Action.none;
189 }
190 break;
191 case Area.leftButton:
192 if (activity == Action.result)
193 {
194 displaying = 0;
195 activity = Action.none;
196 }
197 break;
198 case Area.rightButton:
199 if (activity == Action.result)
200 {
201 activity = Action.none;
202 this.LoadStage();
203 }
204 break;
205 case Area.none:
206 break;
207 default:
208 break;
209 }
210 }

235 public void UpdateLevel(Loop loop)
236 { // odświeżanie danych kolejnego cyklu przez wykonanie obliczeń
237 // charakterystycznych do procesu, który jest aktualnie wykonywany
238 // w trakcie rozgrywki
239 if (MouseEventsGame.CheckArea(loop) == Area.theory)
240 this.ellipse.ChangeFillColor(DisplayLevel.theoryFillClrActive);
241 else
242 this.ellipse.ChangeFillColor(DisplayLevel.theoryFillClr);
243 // warunek spełniony, gdy istnieje rezultat strzału
244 if (stage.updatedResult != 0 && stage.stageNumber == currentStage)
245 {
246 if (stage.updatedResult == 1)
247 { // zwiększanie numeru etapu w razie sukcesu
248 // jeśli ostatni poziom, to zapisz postęp w pliku
249 if (stage.stageNumber == 3)
250 Progress.SaveProgress((short)levelNumber);
251 currentStage = stage.stageNumber + 1;
252 }
253 activity = Level.Action.result;
254 }
255 // jeśli brak jeszcze wyniku, to odśwież obliczenia charakterystyczne dla etapu
256 else if (!(activity == Action.result && stage.updatedResult == 0))
257 stage.UpdateStage(ref activity, loop);
258
