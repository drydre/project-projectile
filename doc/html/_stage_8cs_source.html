300 public void UpdateString(Loop loop)
301 {
302 Vector2i mousePos = Mouse.GetPosition(loop.Window);
303 float length = (float)Math.Sqrt(Math.Pow(slingshotX - mousePos.X, 2) + Math.Pow(slingshotY - mousePos.Y, 2));
304 float alpha = (float)Math.Asin((slingshotY - mousePos.Y) / length);
305 // ograniczenie przed strzelaniem w przeciwnym kierunku (kąt maksymalny +/-90 st.)
306 if (slingshotX - mousePos.X < 0)
307 {
308 alpha = (slingshotY - mousePos.Y <= 0) ? (float)-Math.PI / 2 : (float)Math.PI / 2;
309 mousePos.X = (int)slingshotX;
310 }
311 // ograniczenie rozciągania cięciwy
312 if (length > stringLimit)
313 {
314 length = stringLimit;
315 mousePos.X = (int)(slingshotX - Math.Cos(alpha) * length);
316 mousePos.Y = (int)(slingshotY - Math.Sin(alpha) * length);
317 }
318 // zmienne przypisywane nadmiarowo służą do wyznaczenia ruchu cięciwy w trakcie powrotu
319 this.length = length;
320 this.str_prev = length;
321 this.stretch = length;
322 this.angle = alpha;
323 // wyznaczenie energii dla obecnego naciągnięcia
324 this.energy = physics.GetEnergy(stringK, stretch);
325 //obliczanie skalowania i nachylenia dla poszczególnych fragmentów cięciwy
326 float Y = slingshotY - mousePos.Y;
327 float leftX = slingshotX - slingTexture.Size.X / 2 + C.STRING_DISTANCE - mousePos.X;
328 float leftLength = (float)Math.Sqrt(Math.Pow(leftX, 2) + Math.Pow(Y, 2));
329 float rightX = slingshotX + slingTexture.Size.X / 2 - C.STRING_DISTANCE - mousePos.X;
330 float rightLength = (float)Math.Sqrt(Math.Pow(rightX, 2) + Math.Pow(Y, 2));
331 float leftAlpha = (Y < 0) ? -(float)(Math.Acos(leftX / leftLength) + Math.PI) : -(float)(Math.PI - Math.Acos(leftX / leftLength));
332 float rightAlpha = -(float)(Math.PI - Math.Asin(Y / rightLength));
333 float leftScale = leftLength / stringLeftShape.Size.X;
334 float rightScale = rightLength / stringRightShape.Size.X;
335 // przypisanie nowych skali i pozycji pocisku
336 stringLeftShape.Scale = new Vector2f(leftScale, 1);
337 stringLeftShape.Rotation = leftAlpha * 180 / (float)Math.PI;
338 stringRightShape.Scale = new Vector2f(rightScale, 1);
339 stringRightShape.Rotation = rightAlpha * 180 / (float)Math.PI;
340 projectileShape.Position = (Vector2f)mousePos;
341 }

344 private void CreatePlatforms()
345 {
346 switch (stageNumber)
347 {
348 case 1:
349 this.slingshotPlatformHeight = 0;
350 this.targetPlatformHeight = 0;
351 break;
352 case 2:
353 // -150 / -250 ogranicza wysokość platformy, aby nie nachodzić na obszary stałych
354 // ani przycisku wyświetlania teorii
355 // wysokość platformy zmieniana jest co 25px czyli co 0,5m przy programowym przeliczniku 50 px/m
356 this.targetPlatformHeight = (int)((random.Next((int)((C.DEFAULT_HEIGHT - baseAlt-150 - targetTexture.Size.Y) / C.DISTANCE_SCALE*2)) + 1) * C.DISTANCE_SCALE / 2);
357 this.slingshotPlatformHeight = 0;
358 break;
359 case 3:
360 this.slingshotPlatformHeight = (int)((random.Next((int)((C.DEFAULT_HEIGHT - baseAlt - 250 - slingTexture.Size.Y) / C.DISTANCE_SCALE * 2)) + 1) * C.DISTANCE_SCALE / 2);
361 do
362 {
363 this.targetPlatformHeight = (int)((random.Next((int)((C.DEFAULT_HEIGHT - baseAlt - 150 - targetTexture.Size.Y) / C.DISTANCE_SCALE * 2)) + 1) * C.DISTANCE_SCALE / 2);
364 } while (this.slingshotPlatformHeight == this.targetPlatformHeight);
365 break;
366 default:
367 break;
368 }

375 private void GenerateConsts()
376 {
377 this.mass = (random.Next(9)) / 2f + 1;
378 do

391 private bool CheckK()
392 {
393 float diffAngle = (float)Math.Atan2((int)(targetPlatformHeight - slingshotPlatformHeight - slingTexture.Size.Y + targetTexture.Size.Y), (int)(targetX - slingshotX));
394 float checkLowAngle = diffAngle + C.ANGLE_CUTOFF * (float)Math.PI / 180;
395 float checkHighAngle = diffAngle + C.ANGLE_SAFE_K * (float)Math.PI / 180;
396 // podstawienie skrajnych danych do wzorów opisujących ruch poziomy
397 float tLow = (targetX - slingshotX) / (float)(Math.Cos(checkLowAngle) * Math.Sqrt(stringK * Math.Pow(stringLimit, 2) / mass));
398 float tSafe = (targetX - slingshotX) / (float)(Math.Cos(checkHighAngle) * Math.Sqrt(stringK * Math.Pow(stringLimit, 2) / mass));
399 float yLow = (float)(Math.Sqrt(stringK * Math.Pow(stringLimit, 2) / mass) * Math.Sin(checkLowAngle) * tLow - C.DISTANCE_SCALE * g * Math.Pow(tLow, 2) / 2);
400 float ySafe = (float)(Math.Sqrt(stringK * Math.Pow(stringLimit, 2) / mass) * Math.Sin(checkHighAngle) * tSafe - C.DISTANCE_SCALE * g * Math.Pow(tSafe, 2) / 2);
401 if (yLow + slingshotPlatformHeight + (int)slingTexture.Size.Y >= (float)targetPlatformHeight || ySafe + slingshotPlatformHeight + (int)slingTexture.Size.Y < (float)(targetPlatformHeight + targetTexture.Size.Y ))
402 {
403 return false;
404 }
405 else
406 return true;
407 }

414 private void ReturnString(float dt)
415 {
416 // poniższy wzór to przybliżona różniczka zmiany naciągu po czasie - równanie różnicowe
417 // wyprowadzone z energii kx2^2/2 - kx1^2/2 = mv^2/2, gdzie v = ds/dt, a x1, x2 to naciągi w punkcie początkowym i aktualnym
418 // widać tu uproszczenie kx^2/2 nie jest energią naciągu, ponieważ pominięta zostaje strata na uniesienie
419 // pocisku z powrotem do pozycji wylotowej
420 stretch = ((mass * str_prev - dt * (float)Math.Sqrt(stringK * (stringK * (float)Math.Pow(length, 2) * (float)Math.Pow(dt, 2) - mass * (float)Math.Pow(str_prev, 2) + mass * (float)Math.Pow(length, 2)))) / (stringK * (float)Math.Pow(dt, 2) + mass));
421 if (stretch<0)
422 stretch = 0;
423 // przepisywanie poprzedniego naciągu do wykorzystania w równaniu różnicowym
424 this.str_prev = stretch;
425 Vector2f projectilePos = new Vector2f(slingshotX - (float)Math.Cos(angle) * stretch, slingshotY - (float)Math.Sin(angle) * stretch);
426 // gdyby wyliczona kolejna pozycja procy i gumy wychodziła za punkt wylotu, ustaw pozycję na punkt wylotu
427 if (projectilePos.X > slingshotX)
428 projectilePos = new Vector2f(slingshotX, slingshotY);
429 // obliczenia długości i kątów nachylenia poszczególnych połówek cięciwy procy
430 float Y = slingshotY - projectilePos.Y;
431 float leftX = slingshotX - slingTexture.Size.X / 2 + C.STRING_DISTANCE - projectilePos.X;
432 float leftLength = (float)Math.Sqrt(Math.Pow(leftX, 2) + Math.Pow(Y, 2));
433 float rightX = slingshotX + slingTexture.Size.X / 2 - C.STRING_DISTANCE - projectilePos.X;
434 float rightLength = (float)Math.Sqrt(Math.Pow(rightX, 2) + Math.Pow(Y, 2));
435 float leftAlpha = (Y < 0) ? -(float)(Math.Acos(leftX / leftLength) + Math.PI) : -(float)(Math.PI - Math.Acos(leftX / leftLength));
436 float rightAlpha = -(float)(Math.PI - Math.Asin(Y / rightLength));
437 float leftScale = leftLength / stringLeftShape.Size.X;
438 float rightScale = rightLength / stringRightShape.Size.X;
439 // skalowanie i obracanie połówek cięciwy wg powyższych wyliczeń
440 stringLeftShape.Scale = new Vector2f(leftScale, 1);
441 stringLeftShape.Rotation = leftAlpha * 180 / (float)Math.PI;
442 stringRightShape.Scale = new Vector2f(rightScale, 1);
443 stringRightShape.Rotation = rightAlpha * 180 / (float)Math.PI;
444 projectileShape.Position = projectilePos;
445 }

448 private int CheckHit()
449 {
450 Vector2f projectilePosition = projectileShape.Position;
451 Vector2f targetSize = (Vector2f)targetTexture.Size;
452 if (projectilePosition.X >= targetSprite.Position.X && projectilePosition.X <= targetSprite.Position.X + targetSize.X &&
453 projectilePosition.Y >= targetSprite.Position.Y && projectilePosition.Y <= targetSprite.Position.Y + targetSize.Y)
454 { // przypadek trafienia - głośność zależna od energii strzału i planety
455 Level.hitSound.Volume = 100 * (float)Math.Sqrt(g / C.G_KEPLER) * energy / maxEnergy;
456 Level.hitSound.Play();
457 return 1;
458 }
459 else if (projectilePosition.X > targetSprite.Position.X + targetSize.X || projectilePosition.Y > C.DEFAULT_HEIGHT - baseAlt)
460 { // przypadek wylecenia pocisku poza obszar, w którym możliwe jest trafienie
461 if (projectilePosition.Y > C.DEFAULT_HEIGHT - baseAlt)
462 { // upadek na ziemię - odtworzenie dźwięku
463 Level.missSound.Volume = 100 * (float)Math.Sqrt(g / C.G_KEPLER) * energy / maxEnergy;
464 Level.missSound.Play();
465 }
466 return -1;
467 }
468 // pocisk ciągle w trakcie lotu
469 else
470 return 0;
471 }

612 public void UpdateStage(ref Level.Action activity, Loop loop)
613 {
614 if (activity == Level.Action.result)
615 { // w przypadku czynności wyświetlania rezultatu wczytaj
616 // go z odpowiednimi danymi
617 if (updatedResult == -1)
618 success = false;
619 else if (updatedResult == 1)
620 success = true;
621 LoadResult(success);
622 updatedResult = 0;
623 }
624 else if (activity == Level.Action.flight)
625 { // w przypadku lotu pocisku sprawdzaj, czy dostępny
626 // jest wynik strzału
627 if (updatedResult == 0)
628 {
629 updatedResult = CheckHit();
630 projectileShape.Position = physics.UpdateCoords();
631 UpdateFlightData();
632 }
633 else if (updatedResult != 0)
634 { // wyłączenie zegara stanowiącego licznik
635 // czasu w module fizyki
636 physics.clock.Dispose();
637 }
638 }
639
640 else if (activity == Level.Action.acc)
641 {
642 if (!clockRun)
643 { // zegar służy do obliczeń zachowania cięciwy
644 // i włączany jest tylko raz
645 clock = new Clock();
646 clockRun = true;
647 t_prev = clock.ElapsedTime.AsSeconds();
648 }
649 else if (projectileShape.Position != slingSprite.Position && clockRun)
650 { // odświeżenie długości powracającej cięciwy z przekazaniem
651 // przyrostu czasu do równiania różnicowego
652 ReturnString(clock.ElapsedTime.AsSeconds() - t_prev);
653 t_prev = clock.ElapsedTime.AsSeconds();
654 }
655 else
656 { // wyłączenie zegara po powrocie cięciwy do puntu zerowego
657 // z przejściem do procesu lotu
658 clock.Dispose();
659 clockRun = false;
660 activity = Level.Action.flight;
661 // inicjalizacja fizyki z podaniem parametrów ustalonych przez gracza
662 // oraz stałych charakterystycznych dla obecnego poziomu
663 physics.InitShot(-angle, energy, mass, projectileShape.Position, g);
664 }
665 }
666 else if (activity == Level.Action.aiming)
667 { // w procesie celowania odświeżaj położenie cięciwy
668 // i pocisku dopóki wciśnięty jest lewy przycisk myszy
669 // a także aktualizuj dane strzału wyświetlane obok procy
670 if (Mouse.IsButtonPressed(Mouse.Button.Left))
671 {
672 UpdateShotData();
673 UpdateString(loop);
